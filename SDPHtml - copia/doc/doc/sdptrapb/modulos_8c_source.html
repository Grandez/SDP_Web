357  }

360  asi que se crea un parrafo virtual mas
361  */
362  curr->lstParrs = (TIMER *) malloc(sizeof(TIMER) * (nParrs + 1));
363  for (idx = 0; idx < (nParrs + 1); idx++) {

372  memset(curr->parrs, 0x0, nParrs * sizeof(unsigned long));
373  }
