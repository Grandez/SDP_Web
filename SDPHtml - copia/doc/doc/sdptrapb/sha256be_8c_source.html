7 /*
8  * Description:
9  * This file implements the Secure Hash Algorithms SHA-224 and
10  * SHA-256 as defined in the U.S. National Institute of Standards
11  * and Technology Federal Information Processing Standards
12  * Publication (FIPS PUB) 180-3 published in October 2008
13  * and formerly defined in its predecessors, FIPS PUB 180-1
14  * and FIP PUB 180-2.
15  *
16  * A combined document showing all algorithms is available at
17  * http://csrc.nist.gov/publications/fips/
18  * fips180-3/fips180-3_final.pdf
19  *
20  * The SHA-224 and SHA-256 algorithms produce 224-bit and 256-bit
21  * message digests for a given data stream. It should take about
22  * 2**n steps to find a message with the same digest as a given
23  * message and 2**(n/2) to find any two messages with the same
24  * digest, when n is the digest size in bits. Therefore, this
25  * algorithm can serve as a means of providing a
26  * "fingerprint" for a message.
27  *
28  * Portability Issues:
29  * SHA-224 and SHA-256 are defined in terms of 32-bit "words".
30  * This code uses <stdint.h> (included via "sha.h") to define 32-
31  * and 8-bit unsigned integer types. If your C compiler does not
32  * support 32-bit unsigned integers, this code is not
33  * appropriate.
34  *
35  * Caveats:
36  * SHA-224 and SHA-256 are designed to work with messages less
37  * than 2^64 bits long. This implementation uses SHA224/256Input()
38  * to hash the bits that are a multiple of the size of an 8-bit
39  * octet, and then optionally uses SHA224/256FinalBits()
40  * to hash the final few bits of the input.
41  */

125 /*
126  * SHA256Input
127  *
128  * Description:
129  * This function accepts an array of octets as the next portion
130  * of the message.
131  *
132  * Parameters:
133  * context: [in/out]
134  * The SHA context to update.
135  * message_array[ ]: [in]
136  * An array of octets representing the next portion of
137  * the message.
138  * length: [in]
139  * The length of the message in message_array.
140  *
141  * Returns:
142  * sha Error Code.
143  */
144 int SHA256Input(SHA256Context *context, const uint8_t *message_array, unsigned int length) {
145  if (!context) return shaNull;
146  if (!length) return shaSuccess;
147  if (!message_array) return shaNull;
148  if (context->Computed) return context->Corrupted = shaStateError;
149  if (context->Corrupted) return context->Corrupted;
150 
151  while (length--) {
152  context->Message_Block[context->Message_Block_Index++] =
153  *message_array;
154 
155  if ((SHA256AddLength(context, 8) == shaSuccess) &&
156  (context->Message_Block_Index == SHA256_Message_Block_Size))
157  SHA256ProcessMessageBlock(context);
158 
159  message_array++;
160  }
161 
162  return context->Corrupted;
163 
164 }

166 /*
167  * SHA256FinalBits
168  *
169  * Description:
170  * This function will add in any final bits of the message.
171  *
172  * Parameters:
173  * context: [in/out]
174  * The SHA context to update.
175  * message_bits: [in]
176  * The final bits of the message, in the upper portion of the
177  * byte. (Use 0b###00000 instead of 0b00000### to input the
178  * three bits ###.)
179  * length: [in]
180  * The number of bits in message_bits, between 1 and 7.
181  *
182  * Returns:
183  * sha Error Code.
184  */
185 int SHA256FinalBits(SHA256Context *context, uint8_t message_bits, unsigned int length) {
186  static uint8_t masks[8] = {
187  /* 0 0b00000000 */ 0x00, /* 1 0b10000000 */ 0x80,
188  /* 2 0b11000000 */ 0xC0, /* 3 0b11100000 */ 0xE0,
189  /* 4 0b11110000 */ 0xF0, /* 5 0b11111000 */ 0xF8,
190  /* 6 0b11111100 */ 0xFC, /* 7 0b11111110 */ 0xFE
191  };
192  static uint8_t markbit[8] = {
193  /* 0 0b10000000 */ 0x80, /* 1 0b01000000 */ 0x40,
194  /* 2 0b00100000 */ 0x20, /* 3 0b00010000 */ 0x10,
195  /* 4 0b00001000 */ 0x08, /* 5 0b00000100 */ 0x04,
196  /* 6 0b00000010 */ 0x02, /* 7 0b00000001 */ 0x01
197  };
198 
199  if (!context) return shaNull;
200  if (!length) return shaSuccess;
201  if (context->Corrupted) return context->Corrupted;
202  if (context->Computed) return context->Corrupted = shaStateError;
203  if (length >= 8) return context->Corrupted = shaBadParam;
204 
205  SHA256AddLength(context, length);
206  SHA256Finalize(context, (uint8_t)
207  ((message_bits & masks[length]) | markbit[length]));
208 
209  return context->Corrupted;
210 }

254 void SHA256ProcessMessageBlock(SHA256Context *context) {
255  /* Constants defined in FIPS 180-3, section 4.2.2 */
256  static const uint32_t K[64] = {
257  0x428a2f98, 0x71374491, 0xb5c0fbcf, 0xe9b5dba5, 0x3956c25b,

261  0x2de92c6f, 0x4a7484aa, 0x5cb0a9dc, 0x76f988da, 0x983e5152,
262  0xa831c66d, 0xb00327c8, 0xbf597fc7, 0xc6e00bf3, 0xd5a79147,
263  0x06ca6351, 0x14292967, 0x27b70a85, 0x2e1b2138, 0x4d2c6dfc,
264  0x53380d13, 0x650a7354, 0x766a0abb, 0x81c2c92e, 0x92722c85,
265  0xa2bfe8a1, 0xa81a664b, 0xc24b8b70, 0xc76c51a3, 0xd192e819,
266  0xd6990624, 0xf40e3585, 0x106aa070, 0x19a4c116, 0x1e376c08,
267  0x2748774c, 0x34b0bcb5, 0x391c0cb3, 0x4ed8aa4a, 0x5b9cca4f,
268  0x682e6ff3, 0x748f82ee, 0x78a5636f, 0x84c87814, 0x8cc70208,
269  0x90befffa, 0xa4506ceb, 0xbef9a3f7, 0xc67178f2
270  };
271  int t, t4; /* Loop counter */
272  uint32_t temp1, temp2; /* Temporary word value */
273  uint32_t W[64]; /* Word sequence */
274  uint32_t A, B, C, D, E, F, G, H; /* Word buffers */
275 
276  /*
277  * Initialize the first 16 words in the array W
278  */
279  for (t = t4 = 0; t < 16; t++, t4 += 4)
280  W[t] = (((uint32_t)context->Message_Block[t4]) << 24) |
281  (((uint32_t)context->Message_Block[t4 + 1]) << 16) |
282  (((uint32_t)context->Message_Block[t4 + 2]) << 8) |
283  (((uint32_t)context->Message_Block[t4 + 3]));
284 
285  for (t = 16; t < 64; t++)
286  W[t] = SHA256_sigma1(W[t-2]) + W[t-7] +
287  SHA256_sigma0(W[t-15]) + W[t-16];
288 
289  A = context->Intermediate_Hash[0];
290  B = context->Intermediate_Hash[1];
291  C = context->Intermediate_Hash[2];
292  D = context->Intermediate_Hash[3];
293  E = context->Intermediate_Hash[4];
294  F = context->Intermediate_Hash[5];
295  G = context->Intermediate_Hash[6];
296  H = context->Intermediate_Hash[7];
297 
298  for (t = 0; t < 64; t++) {
299  temp1 = H + SHA256_SIGMA1(E) + SHA_Ch(E,F,G) + K[t] + W[t];
300  temp2 = SHA256_SIGMA0(A) + SHA_Maj(A,B,C);
301  H = G;
302  G = F;
303  F = E;
304  E = D + temp1;
305  D = C;
306  C = B;
307  B = A;
308  A = temp1 + temp2;
309  }
310 
311  context->Intermediate_Hash[0] += A;
312  context->Intermediate_Hash[1] += B;
313  context->Intermediate_Hash[2] += C;
314  context->Intermediate_Hash[3] += D;
315  context->Intermediate_Hash[4] += E;
316  context->Intermediate_Hash[5] += F;
317  context->Intermediate_Hash[6] += G;
318  context->Intermediate_Hash[7] += H;
319 
320  context->Message_Block_Index = 0;
321 }

352 /*
353  * SHA224_256PadMessage
354  *
355  * Description:
356  * According to the standard, the message must be padded to the next
357  * even multiple of 512 bits. The first padding bit must be a '1'.
358  * The last 64 bits represent the length of the original message.
359  * All bits in between should be 0. This helper function will pad
360  * the message according to those rules by filling the
361  * Message_Block array accordingly. When it returns, it can be
362  * assumed that the message digest has been computed.
363  *
364  * Parameters:
365  * context: [in/out]
366  * The context to pad.
367  * Pad_Byte: [in]
368  * The last byte to add to the message block before the 0-padding
369  * and length. This will contain the last bits of the message
370  * followed by another single bit. If the message was an
371  * exact multiple of 8-bits long, Pad_Byte will be 0x80.
372  *
373  * Returns:
374  * Nothing.
375  */
376 void SHA256PadMessage(SHA256Context *context, uint8_t Pad_Byte) {
377 
378  /*
379  * Check to see if the current message block is too small to hold
380  * the initial padding bits and length. If so, we will pad the
381  * block, process it, and then continue padding into a second
382  * block.
383  */
384  if (context->Message_Block_Index >= (SHA256_Message_Block_Size-8)) {
385  context->Message_Block[context->Message_Block_Index++] = Pad_Byte;
386  while (context->Message_Block_Index < SHA256_Message_Block_Size)
387  context->Message_Block[context->Message_Block_Index++] = 0;
388  SHA256ProcessMessageBlock(context);
389  } else
390  context->Message_Block[context->Message_Block_Index++] = Pad_Byte;
391 
392  while (context->Message_Block_Index < (SHA256_Message_Block_Size-8))
393  context->Message_Block[context->Message_Block_Index++] = 0;
394 
395  /*
396  * Store the message length as the last 8 octets
397  */
398  context->Message_Block[56] = (uint8_t)(context->Length_High >> 24);
399  context->Message_Block[57] = (uint8_t)(context->Length_High >> 16);
400  context->Message_Block[58] = (uint8_t)(context->Length_High >> 8);
401  context->Message_Block[59] = (uint8_t)(context->Length_High);
402  context->Message_Block[60] = (uint8_t)(context->Length_Low >> 24);
403  context->Message_Block[61] = (uint8_t)(context->Length_Low >> 16);
404  context->Message_Block[62] = (uint8_t)(context->Length_Low >> 8);
405  context->Message_Block[63] = (uint8_t)(context->Length_Low);
406 
407  SHA256ProcessMessageBlock(context);
408 }

432  int i;
433 
434  if (!context) return shaNull;

436  if (context->Corrupted) return context->Corrupted;

440

444
